def somma ( double a , b ;) : double op;
{ 	
	op = a + b ;
};

def moltiplicazione ( double a , b ;) : double op;
{ 	
	int i;
	i = 0;
	op = 0;
	
	while(i < b) do {
		op = op + a;
		i = i + 1;
	};
};

def divisione ( double a , b ;) : double op;
{ 	
	op = a / b ;
};

def potenza ( double a , b ;) : double op;
{ 	
	int i;
	i = 0;
	op = 1;
	
	while(i < b) do {
		op = op * a;
		i = i + 1;
	};
};

def fibonacci ( double a ;) : double op;
{ 	
	int i, f1, f2;	
	i = 0;
	f1 = 0;
	f2 = 1;
	
	if(a == 0)
		then{
			op = f1;
		};
		
	if(a == 1)
		then{
			op = f2;
		};
		
	while(i < a) do {
		op = f2 + f1;
		f1 = f2;
		f2 = op;
		i = i + 1;
	};
};
